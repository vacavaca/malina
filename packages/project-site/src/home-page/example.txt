import { h, view, mount } from 'malina'

const state = {
  counter: 0
}

const actions = {
  click: () => state => ({
    counter: state.counter + 1
  })
}

const App = view((state, actions) => (
  <div> 
    <p>Clicked: {state.counter} times!</p>
    <button onClick={actions.click}>Click me</button>
  </div>
), state, actions)

mount(document.body, <App />)
